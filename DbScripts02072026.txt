CREATE PROCEDURE [dbo].[USP_UpdateCounterFoilEditedData]
(
      @ID BIGINT,
      @SubjectCode VARCHAR(50),
      @BookletSlNo VARCHAR(50),
      @ModifiedUserID INT
)
AS
BEGIN

    SET NOCOUNT ON;
    
    BEGIN TRY

        IF EXISTS
        (
            SELECT 1
            FROM CounterFoilEditedData
            WHERE ID=@ID
        )
        BEGIN

             INSERT INTO CounterFoilEditedDataLog ([id]
                  ,[filename]
                  ,[barcode]
                  ,[bubble_regno]
                  ,[handwritten_regno]
                  ,[final_regno]
                  ,[discrepancy]
                  ,[discrepancy_detail]
                  ,[candidate_signed]
                  ,[invigilator_signed]
                  ,[subject_code]
                  ,[BookletSlNo]
                  ,[created_at]
                  ,[omr_threshold]
                  ,[whitenerflag]
                  ,[isblack]
                  ,[ModifiedDate]
                  ,[ModifiedUserID])
             
             SELECT
                 [id],[filename]
                  ,[barcode]
                  ,[bubble_regno]
                  ,[handwritten_regno]
                  ,[final_regno]
                  ,[discrepancy]
                  ,[discrepancy_detail]
                  ,[candidate_signed]
                  ,[invigilator_signed]
                  ,[subject_code]
                  ,[BookletSlNo]
                  ,[created_at]
                  ,[omr_threshold]
                  ,[whitenerflag]
                  ,[isblack]
                  ,[ModifiedDate]
                  ,[ModifiedUserID]
             FROM CounterFoilEditedData
             WHERE ID=@ID
            
             UPDATE CounterFoilEditedData
             SET
                 [subject_code]=@SubjectCode,
                 BookletSlNo=@BookletSlNo,
                 ModifiedDate=GETDATE(),
                 [ModifiedUserID]=@ModifiedUserID
             WHERE ID=@ID

        END
        ELSE
        BEGIN

             INSERT INTO CounterFoilEditedData
             ([id],[filename]
                  ,[barcode]
                  ,[bubble_regno]
                  ,[handwritten_regno]
                  ,[final_regno]
                  ,[discrepancy]
                  ,[discrepancy_detail]
                  ,[candidate_signed]
                  ,[invigilator_signed]
                  ,[subject_code]
                  ,[BookletSlNo]
                  ,[created_at]
                  ,[omr_threshold]
                  ,[whitenerflag]
                  ,[isblack]
                  ,[ModifiedDate]
                  ,[ModifiedUserID]
                  )
             SELECT
                 [id]
                  ,[filename]
                  ,[barcode]
                  ,[bubble_regno]
                  ,[handwritten_regno]
                  ,[final_regno]
                  ,[discrepancy]
                  ,[discrepancy_detail]
                  ,[candidate_signed]
                  ,[invigilator_signed]
                  ,@SubjectCode
                  ,@BookletSlNo
                  ,[created_at]
                  ,[omr_threshold]
                  ,[whitenerflag]
                  ,[isblack]
                  ,GETDATE(), @ModifiedUserID
                                                  
             FROM OMR_Results
             WHERE ID=@ID

        END

    END TRY

    BEGIN CATCH
